204 bool Phylip40State::SaveAlignment(newAlignment* alignment, std::ostream* output, std::string* FileName)
205 {
206  /* Generate output alignment in PHYLIP/PHYLIP 4 format (sequential) */
207 
208  int i, j, maxLongName;
209  string *tmpMatrix;
210 
211  /* Check whether sequences in the alignment are aligned or not.
212  * Warn about it if there are not aligned. */
213  if (!alignment->isAligned) {
214  cerr << endl << "ERROR: Sequences are not aligned. Format (phylip) "
215  << "not compatible with unaligned sequences." << endl << endl;
216  return false;
217  }
218 
219  /* Allocate local memory for generating output alignment */
220  tmpMatrix = new string[alignment->sequenNumber];
221 
222  /* Depending on alignment orientation: forward or reverse. Copy directly
223  * sequence information or get firstly the reversed sequences and then
224  * copy it into local memory */
225  for(i = 0; i < alignment->sequenNumber; i++)
226  tmpMatrix[i] = (!Machine->reverse) ?
227  alignment->sequences[i] :
228  utils::getReverse(alignment->sequences[i]);
229 
230  /* Depending on if short name flag is activated (limits sequence name up to
231  * 10 characters) or not, get maximum sequence name length */
232  maxLongName = PHYLIPDISTANCE;
233  for(i = 0; (i < alignment->sequenNumber) && (!Machine->shortNames); i++)
234  maxLongName = utils::max(maxLongName, alignment->seqsName[i].size());
235 
236  /* Generating output alignment */
237  /* First Line: Sequences Number & Residued Number */
238  (*output) << " " << alignment->sequenNumber << " " << alignment->residNumber << endl;
239 
240  /* First Block: Sequences Names & First 60 residues */
241  for(i = 0; i < alignment->sequenNumber; i++)
242  (*output) << setw(maxLongName + 3) << left << alignment->seqsName[i].substr(0, maxLongName)
243  << tmpMatrix[i].substr(0, 60) << endl;
244  (*output) << endl;
245 
246  /* Rest of blocks: Print 60 residues per each blocks of sequences */
247  for(i = 60; i < alignment->residNumber; i += 60) {
248  for(j = 0; j < alignment->sequenNumber; j++)
249  (*output) << tmpMatrix[j].substr(i, 60) << endl;
250  (*output) << endl;
251  }
252  (*output) << endl;
253 
254  /* Deallocate local memory */
255  delete [] tmpMatrix;
256 
257  return true;
258 }
