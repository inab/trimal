248  // Process input files one by one.
249  ReadWriteBaseState* inState = NULL;
250  ifstream inFileHandler;
251  int format_value = 0;
252  int temp_value = 0;
253 
254  for (std::string inFile : *inFiles)
255  {
256  // Open file.
257  inFileHandler.open(inFile);
258  if (!inFileHandler.is_open())
259  {

261  return;
262  }
263 
264  inState = NULL;
265  format_value = 0;
266  temp_value = 0;
267 
268  // Get format State that can handle this alignment.
269  for(ReadWriteBaseState* state : available_states)

271  temp_value = state -> CheckAlignment(&inFileHandler);
272 
273  if (temp_value > format_value)
274  {
275  temp_value = format_value;
276  inState = state;
277  }
278  }
279 
280  // Check if there is a format State to handle the alignment.

284  inFileHandler.close();

286  }
287 
288  // Load alignment one by one and store it on each of the formats specified.

290 // if (reverse) alignment->setReverseFlag(true);
291  int start, end;
292  inFileHandler.close();
