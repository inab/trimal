38 statisticsGaps::statisticsGaps(string *alignmentMatrix, int species, int aminos, int dataType_) {
39 
40  int i, j;
41  char indet;
42 
43  columnLength = species;
44  columns = aminos;
45  maxGaps = 0;
46  halfWindow = 0;
47  dataType = dataType_;
48 
49  if(dataType == SequenceTypes::DNA)
50  indet = 'X';
51  else
52  indet = 'N';
53 
54  /* Memory allocation for the vectors and its initialization */
55  gapsInColumn = new int[columns];
56  utils::initlVect(gapsInColumn, columns, 0);
57 
58  aminosXInColumn = new int[columns];
59  utils::initlVect(aminosXInColumn, columns, 0);
60 
61  gapsWindow = new int[columns];
62  utils::initlVect(gapsWindow, columns, 0);
63 
64  numColumnsWithGaps = new int[species+1];
65  utils::initlVect(numColumnsWithGaps, columnLength+1, 0);
66 
67  /* Count the gaps and indeterminations of each columns */
68  for(i = 0; i < columns; i++) {
69  for(j = 0; j < columnLength; j++) {
70  if(alignmentMatrix[j][i] == '-')
71  gapsInColumn[i]++;
72  else if(alignmentMatrix[j][i] == indet)
73  aminosXInColumn[i]++;
74  }
75 
76  /* Increase the number of colums with the number of gaps of the last processed column */
77  numColumnsWithGaps[gapsInColumn[i]]++;
78  gapsWindow[i] = gapsInColumn[i];
79  if(gapsWindow[i] > maxGaps) maxGaps = gapsWindow[i];
80  }
81 }

125 | bool statisticsGaps::applyWindow(int) |
126 | |
127 | This method computes for each column's alignment its gapwindows' value. For this purpose, the method uses the |
128 | values that previously has been calculated and the window's size value. |
129 | |
130 ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++*/
131 
132 bool statisticsGaps::applyWindow(int _halfWindow) {
133 
134  int i, j, window;
135 
136  /* If one of this conditions is true, we return FALSE: */
137  /* .- If already exists a previously calculated vector for this window size */
138  /* .- If halfWinSize value is greater than 1/4 of alignment length */
139  if((halfWindow == _halfWindow) || (_halfWindow > columns/4))
140  return false;
141 
142  /* Initializate to 0 the vector that will store the number of gaps of each column */
143  /* and the vector that will store window processing results */
144  utils::initlVect(numColumnsWithGaps, columnLength+1, 0);
145  utils::initlVect(gapsWindow, columns, 0);
146 
147  /* Initializate maximum gaps' number per column value and store the mediumWinSize value in the object. */
148  maxGaps = 0;
149  halfWindow = _halfWindow;
150  window = (2 * halfWindow + 1);
151 
152  /* We calculate some statistics for every column in the alignment,and the maximum gaps' number value */
153  for(i = 0; i < columns; i++) {
154  /* Sum the total number of gaps for the considered window */
155  for(j = i - halfWindow, gapsWindow[i] = 0; j <= i + halfWindow; j++) {
156  if(j < 0)
157  gapsWindow[i] += gapsInColumn[-j];
158  else if(j >= columns)
159  gapsWindow[i] += gapsInColumn[((2 * columns - j) - 2)];
160  else
161  gapsWindow[i] += gapsInColumn[j];
162  }
163 
164  /* Calculate, and round to the nearest integer, the number of gaps for the i column */
165  gapsWindow[i] = utils::roundInt(((double) gapsWindow[i]/window));
166  /* Increase in 1 the number of colums with the same number of gaps than column i */
167  numColumnsWithGaps[gapsWindow[i]]++;
168 
169  /* Update the max number of gaps in the alignment, if neccesary */
170  if(gapsWindow[i] > maxGaps)
171  maxGaps = gapsWindow[i];
172  }
173  return true;
174 }

188 double statisticsGaps::calcCutPoint(float minInputAlignment, float gapThreshold)
189  {
190 
191  /* Method to select the cutting point based on gaps values from the input
192  * alignment. The cutting point is selected as the maximum gaps number allowed
193  * in the output alignment given a minimum percentage of the input alignment
194  * to be kept and a maximum gaps number. In case of both values set different
195  * cutting points, the minimum percentage of the input alignment prevails. */
196 
197  double cuttingPoint_MinimumConserv, cuttingPoint_gapThreshold;
198  int i, acum;
199 
200  /* Calculate the gap number represented by the gaps threshold. This gap number
201  * represents the maximum gap number in any column in the output alignment */
202  cuttingPoint_gapThreshold = (double) columnLength * gapThreshold;
203 
204  /* Compute the minimum columns number to be kept from the input alignment */
205  cuttingPoint_MinimumConserv = utils::roundInt(((double)
206  (columns * minInputAlignment) / 100.0));
207  if(cuttingPoint_MinimumConserv > columns)
208  cuttingPoint_MinimumConserv = columns;
209 
210  /* We look at the number of gaps which allows us to keep the minimum columns
211  * number from the input alignment */
212  for(i = 0, acum = 0; i < columnLength; i++) {
213  acum += numColumnsWithGaps[i];
214  if(acum >= cuttingPoint_MinimumConserv)
215  break;
216  }
217 
218  /* If there is not an exact number for the gaps cutting point, compute such
219  * value as the inmediate superior gap number minus the proportion of columns
220  * necessary to respect the minimum percentage from the input alignment to be
221  * kept */
222  if(numColumnsWithGaps[i])
223  cuttingPoint_MinimumConserv = (double) (i - ((float)
224  (acum - cuttingPoint_MinimumConserv)/numColumnsWithGaps[i]));
225  else
226  cuttingPoint_MinimumConserv = 0;
227 
228  /* Return the maximum gap number of the two computed cutting points. */
229  return (utils::max(cuttingPoint_MinimumConserv, cuttingPoint_gapThreshold));
230 }

233 | int statisticsGaps::calcCutPointMixSlope(void) |
234 | |
235 | This method computes and selects the cut point based on the maximum rate between the first slope ratio between |
236 | gaps' percentage in the columns and alignments' length and the "second" slope (slope between three consecutive |
237 | points using only the first and third of them) ratio between gaps' percentage in the columns and alignments' |
238 | length. |
239 | |
240 ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++*/
241 
242 int statisticsGaps::calcCutPointMixSlope(void) {
243 
244  float delta = 0, maxSlope = -1, *firstSlopeVector, *secondSlopeVector;
245  int prev, pprev, maxIter, row = 1, act = 0, max = 0;
246 
247  /* We build two slope vectors, one vector for the first slope and another one for the second. */
248  firstSlopeVector = new float[maxGaps+1];
249  secondSlopeVector = new float[maxGaps+1];
250 
251  /* We initialize them with -1.0 value and fix the maximum iteractions' number as maximun gaps' number plus 1. */
252  utils::initlVect(firstSlopeVector, maxGaps, -1.0);
253  utils::initlVect(secondSlopeVector, maxGaps, -1.0);
254  maxIter = maxGaps + 1;
255 
256  /* Until to achieve the maximum iteractions' number. */
257  while(act < maxIter) {
258 
259  /* We look for a first point to second slope. */
260  while((numColumnsWithGaps[act]) == 0) act++;
261  pprev = act; if((act+1) >= maxIter) break;
262 
263  /* We look for a first point to first slope. */
264  do { act++; } while((numColumnsWithGaps[act]) == 0);
265  prev = act; if((act+1) >= maxIter) break;
266 
267  /* We look for a second point to first and second slope. */
268  do { act++; } while((numColumnsWithGaps[act]) == 0);
269  if(act >= maxIter) break;
270 
271  /* Calculate the first slope between the earlier previus and previus points. */
272  firstSlopeVector[prev] = ((float) (prev - pprev) / columnLength);
273  firstSlopeVector[prev] /= ((float) numColumnsWithGaps[prev] / columns);
274 
275  /* Calculate the first slope between the previus and current points. */
276  firstSlopeVector[act] = ((float) (act - prev) / columnLength);
277  firstSlopeVector[act] /= ((float) numColumnsWithGaps[act] / columns);
278 
279  /* Calculate the second slope between the earlier previus and current points. */
280  secondSlopeVector[act] = ((float) (act - pprev) / columnLength);
281  secondSlopeVector[act] /= ((float) (numColumnsWithGaps[act] + numColumnsWithGaps[prev]) / columns);
282 
283  /* If the ratio between ... */
284  if((secondSlopeVector[pprev] != -1.0) || (firstSlopeVector[pprev] != -1.0)) {
285 
286  /* .- first slope previus and first slope earlier previus points. */
287  if(firstSlopeVector[pprev] != -1.0) {
288  delta = firstSlopeVector[prev]/firstSlopeVector[pprev];
289  row = pprev;
290  }
291 
292  /* .- first slope current and first slope previus points. */
293  if(delta < (firstSlopeVector[act]/firstSlopeVector[prev])) {
294  delta = firstSlopeVector[act]/firstSlopeVector[prev];
295  row = prev;
296  }
297 
298  /* .- second slope current and second slope earlier previus points. */
299  if(secondSlopeVector[pprev] != -1.0) {
300  if(delta < (secondSlopeVector[act]/secondSlopeVector[pprev])) {
301  delta = secondSlopeVector[act]/secondSlopeVector[pprev];
302  row = pprev;
303  }
304  }
305 
306  /* ... is better that current maxSlope then we modify the maxSlope with the best ratio. */
307  if(delta > maxSlope) {
308  maxSlope = delta;
309  max = row;
310  }
311  }
312  act = prev;
313  }
314 
315  /* We delete the local memory. */
316  delete[] firstSlopeVector;
317  delete[] secondSlopeVector;
318 
319  /* and, finally, we return the cut point. */
320  return max;
321 }

332 int statisticsGaps::calcCutPoint2ndSlope(void) {
333 
334  float maxSlope = -1, *secondSlopeVector;
335  int prev, pprev, maxIter, act = 0, max = 0;
336 
337  /* We build one slope vector and fix the maximum iteractions' number as the gaps'number plus 1. */
338  secondSlopeVector = new float[maxGaps+1];
339  utils::initlVect(secondSlopeVector, maxGaps, -1.0);
340  maxIter = maxGaps + 1;
341 
342  /* Find the lowest number of gaps into the input alignment. If there are few
343  * points, it is possible that lowest number of gaps is returned as the thres
344  * hold. It could happen input alignment does not have columns with no-gaps */
345  for(act = 0, max = 0; numColumnsWithGaps[act] == 0; act++)
346  max = act + 1;
347 
348  act = 0;
349  while(act < maxIter) {
350 
351  /* We look for a first point to second slope. */
352  while((numColumnsWithGaps[act]) == 0)
353  act++;
354  pprev = act;
355  if((act+1) >= maxIter)
356  break;
357 
358  /* We look for a first point to first slope. */
359  do {
360  act++;
361  } while((numColumnsWithGaps[act]) == 0);
362  prev = act;
363  if((act+1) >= maxIter)
364  break;
365 
366  /* We look for a second point to first and second slope. */
367  do {
368  act++;
369  } while((numColumnsWithGaps[act]) == 0);
370  if(act >= maxIter)
371  break;
372 
373  /* Calculate the second slope between the earlier previous and current points. */
374  secondSlopeVector[act] = ((float) (act - pprev) / columnLength);
375  secondSlopeVector[act] /= ((float) (numColumnsWithGaps[act] + numColumnsWithGaps[prev]) / columns);
376 
377  /* If the ratio between second slope current and second slope earlier previous points. */
378  if(secondSlopeVector[pprev] != -1.0) {
379  if((secondSlopeVector[act]/secondSlopeVector[pprev]) > maxSlope) {
380  maxSlope = (secondSlopeVector[act]/secondSlopeVector[pprev]);
381  max = pprev;
382  }
383  } else if(secondSlopeVector[prev] != -1.0) {
384  if((secondSlopeVector[act]/secondSlopeVector[prev]) > maxSlope) {
385  maxSlope = (secondSlopeVector[act]/secondSlopeVector[prev]);
386  max = pprev;
387  }
388  }
389  act = prev;
390  }
391 
392  /* We deallocate local memory. */
393  delete[] secondSlopeVector;
394 
395  /* Finally, we return the selected cut point. */
396  return max;
397 }
