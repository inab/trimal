94  void Similarity::calculateMatrixIdentity() {
95  // Create a timerLevel that will report times upon its destruction
96  // which means the end of the current scope.
97  StartTiming("void Similarity::calculateMatrixIdentity() ");
98 
99  // We don't want to calculate the matrix identity
100  // if it has been previously calculated
101  if (matrixIdentity != nullptr)
102  return;
103 
104  // Allocate temporal variables
105  char indet;
106  int i, j, k;
107  int sum, length;
108 
109  // Allocate memory for the matrix identity
110  matrixIdentity = new float *[alig->originalNumberOfSequences];
111  for (i = 0; i < alig->originalNumberOfSequences; i++) {
112  matrixIdentity[i] = new float[alig->originalNumberOfSequences];
113  }
114 
115  // Depending on alignment type, indetermination symbol will be one or other
116  indet = alig->getAlignmentType() & SequenceTypes::AA ? 'X' : 'N';
117 
118  // For each sequences' pair
119  for (i = 0; i < alig->originalNumberOfSequences; i++) {
120  for (j = i + 1; j < alig->originalNumberOfSequences; j++) {
121  // For each position in the alignment of that pair than we are processing
122  for (k = 0, sum = 0, length = 0; k < alig->originalNumberOfResidues; k++) {
123  // If we find a element that is not a gap or an X aminoacid in the first sequence of the pair
124  if ((alig->sequences[i][k] != '-') &&
125  (alig->sequences[i][k] != indet)) {
126 
127  // If we also find a valid element in the second sequence
128  if ((alig->sequences[j][k] != '-') &&
129  (alig->sequences[j][k] != indet))
130  {
131  // If the two valid elements are the same increase the sum
132  if (alig->sequences[j][k] == alig->sequences[i][k])
133  sum++;
134  }
135  // Increase the length of the sequence free of gaps and X elements
136  length++;
137  }
138  // If the first processed element is invalid and in the second we find a valid element increase the length of
139  // the sequence free of gaps and X elements
140  else if ((alig->sequences[j][k] != '-') &&
141  (alig->sequences[j][k] != indet))
142  {
143  length++;
144  }
145  }
146 
147  // Calculate the value of matrix idn for columns j and i
148  matrixIdentity[j][i] = (1.0F - ((float)sum / length) );
149  matrixIdentity[i][j] = matrixIdentity[j][i];
150  }
151  }
152  }

154  bool Similarity::calculateVectors(bool cutByGap) {
155  // Create a timerLevel that will report times upon its destruction
156  // which means the end of the current scope.
157  StartTiming("bool Similarity::calculateVectors(int *gaps) ");
158 
159  // A similarity matrix must be defined. If not, return false
160  if (simMatrix == nullptr)
161  return false;
162 
163  // Calculate the matrix identity in case it's not done before
164  if (matrixIdentity == nullptr)
165  calculateMatrixIdentity();
166 
167  // Create the variable gaps, in case we want to cut by gaps
168  int *gaps = nullptr;
169 
170  // Retrieve the gaps values in case we want to set to 0 the similarity value
171  // in case the gaps value for that column is bigger or equal to 0.8F
172  if (cutByGap)
173  {
174  if (alig->Statistics->gaps == nullptr)
175  alig->Statistics->calculateGapStats();
176  gaps = alig->Statistics->gaps->getGapsWindow();
177  }
178 
179  // Initialize the variables used
180  int i, j, k;
181  float num, den;
182 
183  // Depending on alignment type, indetermination symbol will be one or other
184  char indet = alig->getAlignmentType() & SequenceTypes::AA ? 'X' : 'N';
185 
186  // Q temporal value
187  float Q;
188  // Temporal chars that will contain the residues to compare by pair.
189  char chA, chB;
190 
191  // Calculate the maximum number of gaps a column can have to calculate it's
192  // similarity
193  float gapThreshold = 0.8F * alig->numberOfResidues;
194 
195  // For each column calculate the Q value and the MD value using an equation
196  for (i = 0; i < alig->originalNumberOfResidues; i++) {
197  // Set MDK for columns with gaps values bigger or equal to 0.8F
198  if (cutByGap && gaps[i] >= gapThreshold) {
199  MDK[i] = 0.F;
200  continue;
201  }
202  // For each AAs/Nucleotides' pair in the column we compute its distance
203  for (j = 0, num = 0, den = 0; j < alig->originalNumberOfSequences; j++) {
204 
205 
206  // Calculate the upper value of the residue,
207  // to use in simMatrix->getDistance
208  // This is faster than calculating the upper on that method
209  // as this is done before entering the loop
210  // Doing this before checking if the element is indeterminate or gap
211  // allows to check if the indetermination is not capitalized
212  chA = utils::toUpper(alig->sequences[j][i]);
213 
214  // We don't compute the distance if the first element is
215  // a indeterminate (XN) or a gap (-) element.
216  if ((chA == '-') || (chA == indet))
217  continue;
218 
219  for (k = j + 1; k < alig->originalNumberOfSequences; k++) {
220  // We calculate the upper value of the residue,
221  // to use in simMatrix->getDistance
222  // This is equally faster as if it was done inside the method
223  // but to prevent errors, the method doesn't 'upper'
224  // the given chars.
225  // Doing this before checking if the element is indeterminate or gap
226  // allows to check if the indetermination is not capitalized
227  chB = utils::toUpper(alig->sequences[k][i]);
228 
229  // We don't compute the distance if the second element is
230  // a indeterminate (XN) or a gap (-) element
231  if ((chB == '-') || (chB == indet))
232  continue;
233 
234  // We use the identity value for the two pairs and
235  // its distance based on similarity matrix's value.
236  num += matrixIdentity[j][k] * simMatrix->getDistance(chA, chB);
237  den += matrixIdentity[j][k];
238  }
239  }
240 
241  // If we are processing a column with only one AA/nucleotide, MDK = 0
242  if (den == 0)
243  MDK[i] = 0;
244  else
245  {
246  Q = num / den;
247  // If the MDK value is more than 1, we normalized this value to 1.
248  // Only numbers higher than 0 yield exponents higher than 1
249  // Using this we can test if the result is going to be higher than 1.
250  // And thus, prevent calculating the exp.
251  // Take in mind that the Q is negative, so we must test if Q is LESSER
252  // than one, not bigger.
253  if (Q < 0)
254  MDK[i] = 1.F;
255  else
256  MDK[i] = exp(-Q);
257  }
258  }
259 
260  for (i = 0; i < alig->originalNumberOfSequences; i++)
261  delete[] matrixIdentity[i];
262  delete[] matrixIdentity;
263  matrixIdentity = nullptr;
264 
265  return true;
266  }

268  bool Similarity::applyWindow(int halfW) {
269  // Create a timerLevel that will report times upon its destruction
270  // which means the end of the current scope.
271  StartTiming("bool Similarity::applyWindow(int halfW) ");
272 
273  // Calculate the MDK array if it has not been calculated previously
274  if (MDK == nullptr)
275  calculateVectors(true);
276 
277  // Check is the half window value passed is in the valid range
278  if (halfW > alig->originalNumberOfResidues / 4) {
279  debug.report(ErrorCode::SimilarityWindowTooBig);
280  return false;
281  }
282 
283  // If the current half window is the same as the last one, don't do anything
284  if (halfWindow == halfW) return true;
285 
286  // Save the requested half window. This is useful when making a copy of the
287  // alignment, as the window values are not valid anymore but don't want to
288  // calculate them if not needed anymore
289  halfWindow = halfW;
290 
291  // If the half window requested is 0 or a negative number
292  // we simply delete the window values.
293  if (halfW < 1) {
294  delete[] MDK_Window;
295 
296  MDK_Window = nullptr;
297  return true;
298  }
299 
300  // Initialize the values used in the calculation
301  int i, j, window;
302 
303  // Initialize the MDK window array if it's null
304  if (MDK_Window == nullptr)
305  MDK_Window = new float[alig->originalNumberOfResidues + 1];
306 
307  window = 2 * halfWindow + 1;
308 
309  // Do the average window calculations
310  for (i = 0; i < alig->originalNumberOfResidues; i++) {
311  MDK_Window[i] = 0.F;
312  for (j = i - halfWindow; j <= i + halfWindow; j++) {
313  if (j < 0)
314  MDK_Window[i] += MDK[-j];
315  else if (j >= alig->originalNumberOfResidues)
316  MDK_Window[i] += MDK[((2 * alig->originalNumberOfResidues - j) - 2)];
317  else
318  MDK_Window[i] += MDK[j];
319  }
320 
321  // Calculate the average value, by dividing the values
322  MDK_Window[i] = MDK_Window[i] / (float) window;
323  }
324  return true;
325  }

378  double Similarity::calcCutPoint(float baseLine, float conservationPct) {
379  // Create a timerLevel that will report times upon its destruction
380  // which means the end of the current scope.
381  StartTiming("double Similarity::calcCutPoint(float baseLine, float conservationPct) ");
382  // It computes the cutting point based on alignment's similarity values -
383  // the so-called 'similarity'. It also takes into account the minimum percentage
384  // from the input alignment to be kept. Depending on those two values, the
385  // method will select a different cutting-point.
386 
387  double cuttingPoint_MinimumConserv, cuttingPoint_SimilThreshold;
388  int i, highestPos;
389  float *vectAux;
390 
391  vectAux = new float[alig->originalNumberOfResidues];
392 
393  // Sort a copy of the vector containing the similarity values after applying
394  // any windows methods. Take the columns value that it lower than the minimum
395  // similarity threshold set by the user
396  utils::copyVect(getMdkWindowedVector(), vectAux, alig->originalNumberOfResidues);
397  utils::quicksort(vectAux, 0, alig->originalNumberOfResidues - 1);
398 
399  for (i = alig->originalNumberOfResidues - 1; i >= 0; i--)
400  if (vectAux[i] < conservationPct)
401  break;
402  cuttingPoint_SimilThreshold = vectAux[i];
403 
404  // It is possible that due to number casting, we get a number out of the
405  // vector containing the similarity values - it is not reporting an overflow
406  // situation but giving back a 0 when it should be a number equal (or closer)
407  // to 1.
408  highestPos = (int) ((double) (alig->originalNumberOfResidues - 1) * (100.0 - baseLine) / 100.0);
409  highestPos = highestPos < (alig->originalNumberOfResidues - 1) ? highestPos : alig->originalNumberOfResidues - 1;
410  cuttingPoint_MinimumConserv = vectAux[highestPos];
411 
412  delete[] vectAux;
413 
414  // Return the minimum cutting point between the one set by the threshold and
415  // the one set by the minimum percentage of the input alignment to be kept
416  return (cuttingPoint_MinimumConserv < cuttingPoint_SimilThreshold ?
417  cuttingPoint_MinimumConserv : cuttingPoint_SimilThreshold);
418  }
