57  bool Mold::applyWindow(int halfW) {
58  // Calculate the values array if it has not been calculated previously
59  if (values == nullptr)
60  calculate();
61 
62  // Check is the half window value passed is in the valid range
63  if (halfW > residues / 4) {
64  debug.report(ErrorCode::WindowTooBig);
65  return false;
66  }
67 
68  // If the current half window is the same as the last one, don't do anything
69  if (halfWindowApplied == halfW) return true;
70 
71  // If the half window requested is 0 or a negative number
72  // we simply delete the window values.
73  if (halfW < 1) {
74  if (halfWindowApplied > 0)
75  delete[] valuesWindow;
76 
77  valuesWindow = nullptr;
78  return true;
79  }
80 
81  // Initialize the values used in the calculation
82  int i, j;
83 
84  // Initialize the values window array if it's null
85  if (valuesWindow == nullptr)
86  valuesWindow = new float[residues];
87 
88  // Special value. We don't compute the average
89  // if there is no neighbour on one side
90  valuesWindow[0] = values[0];
91  int currentWindowSize = 1;
92 
93  // First loop: While index is lower than the window value,
94  // We apply a window as big as the index
95  for (i = 1; i < halfW + 1; i++) {
96 
97  // Increase the window size by 2, one residue on each side.
98  currentWindowSize += 2;
99 
100  valuesWindow[i] = 0.F;
101 
102  for (j = 0; j < currentWindowSize; j++) {
103  valuesWindow[i] += values[j];
104  }
105 
106  valuesWindow[i] = valuesWindow[i] / currentWindowSize;
107  }
108 
109  // Second loop: Apply requested window size until
110  // i + windowsize > array size
111  {
112  float window = 2 * halfW + 1;
113 
114  for (; i != residues - halfW; i++) {
115 
116  valuesWindow[i] = values[i];
117 
118  for (j = 1; j < halfW + 1; j++) {
119  valuesWindow[i] += values[i + j];
120  valuesWindow[i] += values[i - j];
121  }
122 
123  valuesWindow[i] = valuesWindow[i] / window;
124  }
125  }
126 
127  // Third loop: As index is bigger than the number of residues - halfwindow
128  // Apply a symetric window as big as possible
129  for (; i < residues - 1; i++) {
130 
131  // Decrease the window size by 2, one residue on each side.
132  currentWindowSize -= 2;
133 
134  valuesWindow[i] = values[i];
135 
136  for (j = 1; (i + j) < residues; j++) {
137  valuesWindow[i] += values[i + j];
138  valuesWindow[i] += values[i - j];
139  }
140 
141  // Calculate the average value, by dividing the current value (sumatory)
142  // by the current window size
143  valuesWindow[i] = valuesWindow[i] / currentWindowSize;
144 
145  }
146 
147  // Special value. We don't compute the average
148  // if there is no neighbour on one side
149  valuesWindow[residues - 1] = values[residues - 1];
150 
151  // New window has been applied
152  halfWindowApplied = halfW;
153 
154  return true;
155  }
