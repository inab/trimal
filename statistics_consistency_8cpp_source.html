194 int statisticsConsistency::compareAndChoose(newAlignment **vectAlignments,
195  char **fileNames,
196  float *columnsValue,
197  int numAlignments,
198  bool verbosity) {
199  // Create a timerLevel that will report times upon its destruction
200  // which means the end of the current scope.
201  StartTiming("int statisticsConsistency::compareAndChoose("
202  "newAlignment **vectAlignments, "
203  "char **fileNames, "
204  "float *columnsValue, "
205  "int numAlignments, "
206  "bool verbosity) ");
207 
208  int *numResiduesAlig, *correspNames, *columnSeqMatrix, *columnSeqMatrixAux;
209  int i, j, k, l, m, numSeqs, pairRes, hits, alig = 0;
210  float max = 0, value = 0, **vectHits;
211  bool appearErrors = false;
212  string *names;
213 
214 
215  // Get some parameters from the alignment that has
216  // been selected */
217  numSeqs = vectAlignments[0]->getNumSpecies();
218 
219  // Allocate dinamic local memory
220  names = new string[numSeqs];
221  correspNames = new int[numSeqs];
222  numResiduesAlig = new int[numAlignments];
223  columnSeqMatrix = new int[numSeqs];
224  vectHits = new float *[numAlignments];
225  columnSeqMatrixAux = new int[numSeqs];
226 
227  // Check that all of alignment has the same number of
228  // sequence as well as there exists a correspondence
229  // between the names for each pars of aligs.
230  for (i = 1; i < numAlignments; i++) {
231 
232  if (numSeqs != vectAlignments[i]->getNumSpecies()) {
233  debug.report(ErrorCode::DifferentNumberOfSequencesInCompareset);
234  appearErrors = true;
235  break;
236  }
237 
238  vectAlignments[i]->getSequences(names);
239  if (!vectAlignments[0]->getSequenceNameOrder(names, correspNames)) {
240  debug.report(ErrorCode::DifferentSeqsNamesInCompareset);
241  appearErrors = true;
242  break;
243  }
244  }
245 
246  if (!appearErrors)
247  {
248 
249  // Changes the order in sequences number matrix
250  // according to the order in the selected alignment
251  for (i = 1; ((i < numAlignments) && (!appearErrors)); i++) {
252  vectAlignments[i]->getSequences(names);
253  vectAlignments[0]->getSequenceNameOrder(names, correspNames);
254  vectAlignments[i]->SequencesMatrix->setOrder(correspNames);
255  }
256 
257  // Get back the residues number for each alignment
258  for (i = 0; ((i < numAlignments) && (!appearErrors)); i++)
259  numResiduesAlig[i] = vectAlignments[i]->getNumAminos();
260 
261  // Start the comparison among the alignments
262  for (i = 0; ((i < numAlignments) && (!appearErrors)); i++) {
263  value = 0;
264  // If it's necessary, we print some information
265  if (verbosity)
266  cout << endl;
267 
268  // Initialize the hits vector for each alignment
269  vectHits[i] = new float[numResiduesAlig[i]];
270  utils::initlVect(vectHits[i], numResiduesAlig[i], 0);
271 
272  for (j = 0, pairRes = 0, hits = 0; j < numResiduesAlig[i]; j++, pairRes = 0, hits = 0) {
273 
274 
275  // Get back each column from the current selected
276  // alignment
277  vectAlignments[i]->SequencesMatrix->getColumn(j, columnSeqMatrix);
278 
279  // For each position from the previous recovered
280  // columns, we carry out the analysis to detect the
281  // same residues pair in the rest of the alignmetns
282  for (k = 0; k < numSeqs; k++) {
283 
284  // If there is a valid residue, we go ahead with
285  // the analysis
286  if (columnSeqMatrix[k] != 0) {
287  for (l = 0; l < i; l++) {
288  // Recover the residue pairs from the others aligs
289  vectAlignments[l]->SequencesMatrix->getColumn(columnSeqMatrix[k], k, columnSeqMatrixAux);
290  // and count the similar residue pairs
291  for (m = k + 1; m < numSeqs; m++)
292  if (columnSeqMatrix[m] != 0) {
293  if (columnSeqMatrix[m] == columnSeqMatrixAux[m])
294  hits++;
295  pairRes++;
296  }
297  }
298 
299  for (l = i + 1; l < numAlignments; l++) {
300  // Recover the residue pairs from the others aligs
301  vectAlignments[l]->SequencesMatrix->getColumn(columnSeqMatrix[k], k, columnSeqMatrixAux);
302  // and count the similar residue pairs
303  for (m = k + 1; m < numSeqs; m++)
304  if (columnSeqMatrix[m] != 0) {
305  if (columnSeqMatrix[m] == columnSeqMatrixAux[m])
306  hits++;
307  pairRes++;
308  }
309  }
310  }
311  }
312 
313  // For each column, compute the hits proportion for
314  // every residue pair against the rest of alignments
315  if (pairRes != 0) {
316  vectHits[i][j] += ((1.0 * hits) / pairRes);
317  value += vectHits[i][j];
318  }
319  }
320 
321  // The method can offer some information about the
322  // comparison progression
323  if (verbosity)
324  cout << "File:\t\t" << fileNames[i] << endl << "Values:\t\tSequences: " << numSeqs
325  << "\tResidues: " << numResiduesAlig[i] << "\tPond. Hits: " << setw(8)
326  << value << "\t%Consistency: " << value / numResiduesAlig[i] << endl;
327  // Keep the alignment with higher consistency value
328  if ((value / numResiduesAlig[i]) > max) {
329  alig = i;
330  max = value / numResiduesAlig[i];
331  }
332  }
333 
334  // Prints the alignment that have been selected
335  if ((verbosity) && (!appearErrors)) {
336  cout << "\t\t\t\t\t--------------" << endl;
337  cout << endl << "File Selected:\t" << fileNames[alig] << endl << "Value:\t\t" << max << endl << endl;
338  }
339 
340  // The method returns a vector with the consistency
341  // value for each column in the selected alignment
342  if ((columnsValue != nullptr) && (!appearErrors)) {
343  utils::initlVect(columnsValue, numResiduesAlig[alig], -1);
344  for (i = 0; i < numResiduesAlig[alig]; i++)
345  columnsValue[i] = vectHits[alig][i];
346  }
347  }
348 
349  // Deallocate memory
350  for (i = 0; ((i < numAlignments) && (!appearErrors)); i++)
351  delete[] vectHits[i];
352 
353  delete[] vectHits;
354  delete[] names;
355  delete[] correspNames;
356  delete[] numResiduesAlig;
357  delete[] columnSeqMatrix;
358  delete[] columnSeqMatrixAux;
359 
360  // Return the selected alignment index or an error
361  // flag otherwise
362  if (appearErrors) return -1;
363  else return alig;
364 }

368 bool statisticsConsistency::forceComparison(newAlignment **vectAlignments,
369  int numAlignments,
370  newAlignment *selected,
371  float *columnsValue) {
372  // Create a timerLevel that will report times upon its destruction
373  // which means the end of the current scope.
374  StartTiming("bool statisticsConsistency::forceComparison("
375  "newAlignment **vectAlignments, "
376  "int numAlignments, "
377  "newAlignment *selected, "
378  "float *columnsValue) ");
379 
380  int *correspNames, *columnSeqMatrix, *columnSeqMatrixAux;
381  int i, j, k, ll, numResidues, numSeqs, pairRes, hit;
382  bool appearErrors = false;
383  string *names;
384 
385  // Get some parameters from the alignment that has
386  // been selected
387  numResidues = selected->getNumAminos();
388  numSeqs = selected->getNumSpecies();
389 
390  // Initialize the vector where we are going to store
391  // the proportion of hits for each column in the
392  // selected alignment
393  utils::initlVect(columnsValue, numResidues, 0);
394 
395  // Allocate dinamic local memory
396  names = new string[numSeqs];
397  correspNames = new int[numSeqs];
398  columnSeqMatrix = new int[numSeqs];
399  columnSeqMatrixAux = new int[numSeqs];
400  // Check that all of alignment has the same number of
401  // sequence as well as there exists a correspondence
402  // between the names for each pars of aligs.
403  for (i = 0; i < numAlignments; i++) {
404 
405  if (numSeqs != vectAlignments[i]->getNumSpecies()) {
406  debug.report(ErrorCode::DifferentNumberOfSequencesInCompareset);
407  appearErrors = true;
408  break;
409  }
410 
411  vectAlignments[i]->getSequences(names);
412  if (!selected->getSequenceNameOrder(names, correspNames)) {
413  debug.report(ErrorCode::DifferentSeqsNamesInCompareset);
414  appearErrors = true;
415  break;
416  }
417  }
418  // Changes the order in sequences number matrix
419  // according to the order in the selected alignment
420  for (i = 0; i < numAlignments; i++) {
421  vectAlignments[i]->getSequences(names);
422  selected->getSequenceNameOrder(names, correspNames);
423  vectAlignments[i]->SequencesMatrix->setOrder(correspNames);
424  }
425  // Do the same analysis for each column
426  for (i = 0, pairRes = 0, hit = 0;
427  i < numResidues && !appearErrors;
428  i++, pairRes = 0, hit = 0) {
429 
430  // We get back the sequence position for each residue
431  // from every column in the selected alignment
432  utils::initlVect(columnSeqMatrix, numSeqs, 0);
433  selected->SequencesMatrix->getColumn(i, columnSeqMatrix);
434  // For each residue pairs, we look for it in the rest
435  // of alignments
436  for (j = 0; j < numSeqs; j++) {
437 
438  // If there is a residue, not a gap, we carry out the
439  // assesment
440  if (columnSeqMatrix[j] != 0) {
441  for (k = 0; k < numAlignments; k++) {
442 
443  // We look for the same residue in the same row in
444  // the rest of alignments
445  utils::initlVect(columnSeqMatrixAux, numSeqs, 0);
446  vectAlignments[k]->SequencesMatrix->getColumn(columnSeqMatrix[j], j, columnSeqMatrixAux);
447  // We count when we get the same residue pairs in the
448  // rest of alignments
449  for (ll = j + 1; ll < numSeqs; ll++)
450  if (columnSeqMatrix[ll] != 0) {
451  if (columnSeqMatrix[ll] == columnSeqMatrixAux[ll])
452  hit++;
453  pairRes++;
454  }
455  }
456  }
457  }
458  // Store the hits proportion for each column
459  if (pairRes != 0) columnsValue[i] += ((1.0 * hit) / pairRes);
460  }
461 
462  // Deallocate dynamic memory
463  delete[] names;
464  delete[] correspNames;
465  delete[] columnSeqMatrix;
466  delete[] columnSeqMatrixAux;
467 
468  // Return if the process was perform without errors
469  return !appearErrors;
470

473 // This method applies a specific windows size to a selected alignment
474 bool statisticsConsistency::applyWindow(int _halfWindow) {
475  // Create a timerLevel that will report times upon its destruction
476  // which means the end of the current scope.
477  StartTiming("bool statisticsConsistency::applyWindow(int columns, int halfWindowApplied, float *columnsValue) ");
478 
479  if (_halfWindow > residues / 4)
480  {
481  debug.report(ErrorCode::ConsistencyWindowTooBig);
482  exit(-1);
483  }
484 
485  // If the current half window is the same as the last one, don't do anything
486  if (halfWindow == _halfWindow) return true;
487 
488  // Save the requested half window. This is useful when making a copy of the
489  // alignment, as the window values are not valid anymore but don't want to
490  // calculate them if not needed anymore
491  halfWindow = _halfWindow;
492 
493  // If the half window requested is 0 or a negative number
494  // we simply delete the window values.
495  if (_halfWindow < 1) {
496  delete[] values_windowed;
497  values_windowed = nullptr;
498  return true;
499  }
500 
501  // Initialize the values used in the calculation
502  int i, j, window;
503 
504  // Initialize the consistency window array if it's null
505  if (values_windowed == nullptr)
506  values_windowed = new float[residues];
507 
508 
509  window = 2 * halfWindow + 1;
510 
511  // Do the average window calculations
512  for (i = 0; i < residues; i++) {
513  values_windowed[i] = 0.F;
514  for (j = i - halfWindow; j <= i + halfWindow; j++) {
515  if (j < 0)
516  values_windowed[i] += values[-j];
517  else if (j >= residues)
518  values_windowed[i] += values[((2 * residues - j) - 2)];
519  else
520  values_windowed[i] += values[j];
521  }
522 
523  // Calculate the similarity value for the i column
524  values_windowed[i] = values_windowed[i] / window;
525  }
526 
527  return true;
528 }
