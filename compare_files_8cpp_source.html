39 int compareFiles::algorithm(newAlignment **vectAlignments, char **fileNames, float *columnsValue, int numAlignments, bool verbosity) {
40 
41  int *numResiduesAlig, *correspNames, *columnSeqMatrix, *columnSeqMatrixAux;
42  int i, j, k, l, m, numSeqs, pairRes, hits, alig = 0;
43  float max = 0, value = 0, **vectHits;
44  bool appearErrors = false;
45  string *names;
46 
47 
48  // Get some parameters from the alignment that has
49  // been selected */
50  numSeqs = vectAlignments[0] -> getNumSpecies();
51 
52  // Allocate dinamic local memory
53  names = new string[numSeqs];
54  correspNames = new int[numSeqs];
55  numResiduesAlig = new int[numSeqs];
56  columnSeqMatrix = new int[numSeqs];
57  vectHits = new float*[numAlignments];
58  columnSeqMatrixAux = new int[numSeqs];
59 
60  // Check that all of alignment has the same number of
61  // sequence as well as there exists a correspondence
62  // between the names for each pars of aligs.
63  for(i = 1; i < numAlignments; i++) {
64 
65  if(numSeqs != vectAlignments[i] -> getNumSpecies()) {
66  debug.report(ErrorCode::DifferentNumberOfSequencesInCompareset);
67  appearErrors = true;
68  break;
69  }
70 
71  vectAlignments[i] -> getSequences(names);
72  if(!vectAlignments[0] -> getSequenceNameOrder(names, correspNames)) {
73  debug.report(ErrorCode::DifferentSeqsNamesInCompareset);
74  appearErrors = true;
75  break;
76  }
77  }
78 
79  // Changes the order in sequences number matrix
80  // according to the order in the selected alignment
81  for(i = 1; ((i < numAlignments) && (!appearErrors)); i++) {
82  vectAlignments[i] -> getSequences(names);
83  vectAlignments[0] -> getSequenceNameOrder(names, correspNames);
84  vectAlignments[i] -> SequencesMatrix -> setOrder(correspNames);
85  }
86 
87  // Get back the residues number for each alignment
88  for(i = 0; ((i < numAlignments) && (!appearErrors)); i++)
89  numResiduesAlig[i] = vectAlignments[i] -> getNumAminos();
90 
91  // Start the comparison among the alignments
92  for(i = 0; ((i < numAlignments) && (!appearErrors)); i++, value = 0) {
93 
94  // If it's necessary, we print some information
95  if(verbosity)
96  cout << endl;
97 
98  // Initialize the hits vector for each alignment
99  vectHits[i] = new float[numResiduesAlig[i]];
100  utils::initlVect(vectHits[i], numResiduesAlig[i], 0);
101 
102  for(j = 0, pairRes = 0, hits = 0; j < numResiduesAlig[i]; j++, pairRes = 0, hits = 0) {
103 
104 
105  // Get back each column from the current selected
106  // alignment
107  vectAlignments[i] -> SequencesMatrix -> getColumn(j, columnSeqMatrix);
108 
109  // For each position from the previous recovered
110  // columns, we carry out the analysis to detect the
111  // same residues pair in the rest of the alignmetns
112  for(k = 0; k < numSeqs; k++) {
113 
114  // If there is a valid residue, we go ahead with
115  // the analysis
116  if(columnSeqMatrix[k] != 0) {
117  for(l = 0; l < i; l++) {
118  // Recover the residue pairs from the others aligs
119  vectAlignments[l] -> SequencesMatrix -> getColumn(columnSeqMatrix[k], k, columnSeqMatrixAux);
120  // and count the similar residue pairs
121  for(m = k + 1; m < numSeqs; m++)
122  if(columnSeqMatrix[m] != 0) {
123  if(columnSeqMatrix[m] == columnSeqMatrixAux[m])
124  hits++;
125  pairRes++;
126  }
127  }
128 
129  for(l = i + 1; l < numAlignments; l++) {
130  // Recover the residue pairs from the others aligs
131  vectAlignments[l] -> SequencesMatrix -> getColumn(columnSeqMatrix[k], k, columnSeqMatrixAux);
132  // and count the similar residue pairs
133  for(m = k + 1; m < numSeqs; m++)
134  if(columnSeqMatrix[m] != 0) {
135  if(columnSeqMatrix[m] == columnSeqMatrixAux[m])
136  hits++;
137  pairRes++;
138  }
139  }
140  }
141  }
142 
143  // For each column, compute the hits proportion for
144  // every residue pair against the rest of alignments
145  if(pairRes != 0) {
146  vectHits[i][j] += ((1.0 * hits)/pairRes);
147  value += vectHits[i][j];
148  }
149  }
150 
151  // The method can offer some information about the
152  // comparison progression
153  if(verbosity)
154  cout << "File:\t\t" << fileNames[i] << endl << "Values:\t\tSequences: " << numSeqs
155  << "\tResidues: " << numResiduesAlig[i] << "\tPond. Hits: " << setw(8)
156  << value << "\t%Consistency: " << value/numResiduesAlig[i] << endl;
157  // Keep the alignment with higher consistency value
158  if((value/numResiduesAlig[i]) > max) {
159  alig = i;
160  max = value/numResiduesAlig[i];
161  }
162  }
163 
164  // Prints the alignment that have been selected
165  if((verbosity) && (!appearErrors)) {
166  cout << "\t\t\t\t\t--------------" << endl;
167  cout << endl << "File Selected:\t" << fileNames[alig] << endl << "Value:\t\t" << max << endl << endl;
168  }
169 
170  // The method returns a vector with the consistency
171  // value for each column in the selected alignment
172  if((columnsValue != NULL) && (!appearErrors)) {
173  utils::initlVect(columnsValue, numResiduesAlig[alig], -1);
174  for(i = 0; i < numResiduesAlig[alig]; i++)
175  columnsValue[i] = vectHits[alig][i];
176  }
177  // Deallocate memmory
178  for(i = 0; ((i < numAlignments) && (!appearErrors)); i++)
179  delete [] vectHits[i];
180  delete [] vectHits;
181 
182  delete [] names;
183  delete [] correspNames;
184  delete [] numResiduesAlig;
185  delete [] columnSeqMatrix;
186  delete [] columnSeqMatrixAux;
187 
188  // Return the selected alignment index or an error
189  // flag otherwise
190  if(appearErrors) return -1;
191  else return alig;
192 }

196 bool compareFiles::forceComparison(newAlignment **vectAlignments, int numAlignments, newAlignment *selected, float *columnsValue) {
197 
198  int *correspNames, *columnSeqMatrix, *columnSeqMatrixAux;
199  int i, j, k, ll, numResidues, numSeqs, pairRes, hit;
200  bool appearErrors = false;
201  string *names;
202 
203  // Get some parameters from the alignment that has
204  // been selected
205  numResidues = selected -> getNumAminos();
206  numSeqs = selected -> getNumSpecies();
207 
208  // Initialize the vector where we are going to store
209  // the proportion of hits for each column in the
210  // selected alignment
211  utils::initlVect(columnsValue, numResidues, 0);
212 
213  // Allocate dinamic local memory
214  names = new string[numSeqs];
215  correspNames = new int[numSeqs];
216  columnSeqMatrix = new int[numSeqs];
217  columnSeqMatrixAux = new int[numSeqs];
218  // Check that all of alignment has the same number of
219  // sequence as well as there exists a correspondence
220  // between the names for each pars of aligs.
221  for(i = 0; i < numAlignments; i++) {
222 
223  if(numSeqs != vectAlignments[i] -> getNumSpecies()) {
224  debug.report(ErrorCode::DifferentNumberOfSequencesInCompareset);
225  appearErrors = true;
226  break;
227  }
228 
229  vectAlignments[i] -> getSequences(names);
230  if(!selected -> getSequenceNameOrder(names, correspNames)) {
231  debug.report(ErrorCode::DifferentSeqsNamesInCompareset);
232  appearErrors = true;
233  break;
234  }
235  }
236  // Changes the order in sequences number matrix
237  // according to the order in the selected alignment
238  for(i = 0; i < numAlignments; i++) {
239  vectAlignments[i] -> getSequences(names);
240  selected -> getSequenceNameOrder(names, correspNames);
241  vectAlignments[i] -> SequencesMatrix -> setOrder(correspNames);
242  }
243  // Do the same analysis for each column
244  for(i = 0, pairRes = 0, hit = 0; ((i < numResidues) && (!appearErrors)); i++, pairRes = 0, hit = 0) {
245 
246  // We get back the sequence position for each residue
247  // from every column in the selected alignment
248  utils::initlVect(columnSeqMatrix, numSeqs, 0);
249  selected -> SequencesMatrix -> getColumn(i, columnSeqMatrix);
250  // For each residue pairs, we look for it in the rest
251  // of alignments
252  for(j = 0; j < numSeqs; j++) {
253 
254  // If there is a residue, not a gap, we carry out the
255  // assesment
256  if(columnSeqMatrix[j] != 0) {
257  for(k = 0; k < numAlignments; k++) {
258 
259  // We look for the same residue in the same row in
260  // the rest of alignments
261  utils::initlVect(columnSeqMatrixAux, numSeqs, 0);
262  vectAlignments[k] -> SequencesMatrix -> getColumn(columnSeqMatrix[j], j, columnSeqMatrixAux);
263  // We count when we get the same residue pairs in the
264  // rest of alignments
265  for(ll = j + 1; ll < numSeqs; ll++)
266  if(columnSeqMatrix[ll] != 0) {
267  if(columnSeqMatrix[ll] == columnSeqMatrixAux[ll])
268  hit++;
269  pairRes++;
270  }
271  }
272  }
273  }
274  // Store the hits proportion for each column
275  if(pairRes != 0) columnsValue[i] += ((1.0 * hit)/pairRes);
276  }
277 
278  // Deallocate dinamic memory
279  delete [] names;
280  delete [] correspNames;
281  delete [] columnSeqMatrix;
282  delete [] columnSeqMatrixAux;
283 
284  // If it was fine, return true. Otherwise, return
285  // false
286  if(appearErrors) return false;
287  else return true;
288

291 // This method applies a specific windows size to a selected alignment
292 bool compareFiles::applyWindow(int columns, int halfWindow, float *columnsValue) {
293 
294  int i, j, window;
295  float *vectAux;
296 
297  // If windows size is greater than 1/4 of alignment
298  // length, trimAl rejects this windows size
299  if(halfWindow > columns/4) return false;
300  else window = (2 * halfWindow + 1);
301 
302  // Allocate local memory. Copy the array values to
303  // auxiliar memory
304  vectAux = new float[columns];
305  utils::copyVect(columnsValue, vectAux, columns);
306 
307  // For each column from the selected alignment,
308  // compute the average for its consistency values
309  for(i = 0; i < columns; i++) {
310 
311  // This average is computed from halfWindow positions
312  // before to halfWindow positions after
313  for(j = i - halfWindow, columnsValue[i] = 0; j <= i + halfWindow; j++) {
314  if(j < 0) columnsValue[i] += vectAux[-j];
315  else if(j >= columns)
316  columnsValue[i] += vectAux[((2 * columns - j) - 2)];
317  else columnsValue[i] += vectAux[j];
318  }
319 
320  // Finally, the column value is divided by the window
321  // size in order to compute the average score.
322  columnsValue[i] /= window;
323  }
324 
325  // Deallocate dinamic memory
326  delete [] vectAux;
327 
328  // If everything is OK, return true
329  return true;
330 }
